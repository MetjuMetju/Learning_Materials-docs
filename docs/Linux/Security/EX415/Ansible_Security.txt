# 1. Installation
## Install Ansible
CODE:
sudo dnf install -y ansible-core
ansible --version
ENDCODE

# 2. Inventory
## Verify Inventory
CODE:
cat inventory
ansible all -m ping
ENDCODE

# 3. SELinux Automation
## Apply SELinux Configuration
CODE:
ansible-playbook selinux.yml
ENDCODE

# 4. Firewalld Automation
## Configure Firewall
CODE:
ansible-playbook firewalld.yml
ENDCODE

# 5. OpenSCAP Automation
## Apply Remediation
CODE:
ansible-playbook openscap-remediation.yml
ENDCODE

# 6. Verification
## Check Results
CODE:
ansible all -m setup
ENDCODE
